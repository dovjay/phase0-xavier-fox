## 2. Tahun Kabisat
  STORE "year" with any value
  IF "year" mod 4 equal to 0 AND "year" mod 100 not equal to 0
    DO "Tahun Kabisat"
  ELSE IF "year" mod 4 equal to 0 AND "year" mod 100 equal to 0 AND "year" mod 400 equal to 0
    DO "Tahun Kabisat"
  ELSE
    DO "Bukan Tahun Kabisat"
